شركة زينكو

زينكــو شــركة وطنيــة تأسســت عــام 1994 م بهــدف تقديــم أفضــل و أجــود أنــواع المســتلزمات الخاصــة بالبنــاء و العمــار

و ذلــك مــن خــال ريادتنــا فــي الأســواق المســتهدفة و توفيــر أحــدث المنتجــات المميــزة و الآمنــة و بأعلــى المواصفــات و المعاييــر الدوليــة مــن حيــث كفــاءة الطاقــة و ترشــيد إســتهاك الميــاه

رؤيتنا

أن نكــون مــن أوائــل الشــركات الرائــدة فــي تقديــم منتجــات تلبــي رغبات المســتهلك فــي مجال الإنــارة و الأدوات الصحية و كل مــا يتعلــق بهــا مــن حيــث الســعر و الجــودة و خدمــة مابعــد البيــع و كذلــك مــن خلال تأهيــل الكــوادر الوطنيــة الواعدة

قيمنا

نعتــز فــي زينكــو بقيمنــا فــي التعامــل و التــي نســتمدها مــن تراثنــا العميــق و النظريــات العلميــة الحديثــة و الإنجــازات المتطــورة و ذلــك نلبــي جميــع احتياجاتكــم و نواكــب كل تطلعاتكــم نحو النمو مســتقبل أفضل للجميع .

رسالتنا

نؤمــن فــي شركة زينكــو بــأن توفيــر المنتجات عاليــة الجــودة و موثوقــة و آمنــه و عصريــة ضــرورة حتميــة
وذلــك تماشــيا مــع إحتياجــات العصــر
